package shape;

public class triangle {
	protected double area;
	protected double perimeter;
	public double side1;
	public double side2;
	public double base;
	protected double height;
	
	public triangle(double side1,double side2,double base,double height) {
		this.side1 = side1;
		this.side2 = side2;
		this.base = base;
		this.height = height;
	}
	public double getArea() {
		this.area = (0.5f*this.base*this.height);
		return area;
	}
	public double getPerimeter() {
		this.perimeter = (this.side1+this.side2+this.base);
		return perimeter;
	}

}






package shape;

public class square {
	protected double area;
	private double perimeter;
	protected double side;
	
	public square(double side) {
		this.side = side;
	}
	public double getArea() {
		this.area = (this.side*this.side);
		return area;
	}
	public double getPerimeter() {
		this.perimeter = (4*this.side);
		return perimeter;
	}


}






package shape;

public class circle {
	protected double area;
	double circumference;
	protected double radius;
	
	public circle(double radius) {
		this.radius = radius;
	}
	public double getArea() {
		this.area = (3.14f*this.radius*this.radius);
		return area;
	}
	public double getcircumference() {
		this.circumference = (2*3.14f*this.radius);
		return circumference;
	}


}
